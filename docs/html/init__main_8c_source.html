26/**
27 * @brief Initializes the game window using MLX library
28 *
29 * Creates a new window with dimensions calculated from map
30 * dimensions and tile size.
31 * The window title is set to "so_long". If window creation
32 * fails, performs cleanup
33 * and exits with error message.
34 *
35 * @param game Pointer to the game structure containing MLX
36 * and map information
37 *
38 * @example
39 * t_game game;
40 * game.mlx = mlx_init();
41 * game.map_width = 10;
42 * game.map_height = 8;
43 * game.tile_size = 32;
44 * init_game_window(&game);
45 * // Creates a 320x256 pixel window
46 *
47 * @note Requires valid MLX instance and initialized map dimensions
48 * @warning Will exit program if window creation fails
49 *
50 * @return void
51 */
52void init_game_window(t_game *game)
53{
54 int win_width;
55 int win_height;
56
57 win_width = game->map_width * game->tile_size;
58 win_height = game->map_height * game->tile_size;
59 game->win = mlx_new_window(game->mlx, win_width, win_height, "so_long");
60 if (!game->win)
61 {
62 cleanup_game(game);
63 ft_printf("Error: Failed to create window\n");
64 exit(1);
65 }
66}

68/**
69 * @brief Initializes the MLX library and loads the game map
70 *
71 * This function performs two main tasks:
72 * 1. Initializes the MLX (MinilibX) graphics library
73 * 2. Reads and validates the game map from the specified path
74 *
75 * @param game Pointer to the game structure containing MLX and map data
76 * @param map_path String containing the path to the map file
77 *
78 * @throws Exits with status 1 if MLX initialization fails
79 * @throws Exits with status 1 if map is invalid or cannot be read
80 *
81 * @example
82 * t_game game;
83 * init_mlx_and_map(&game, "./maps/level1.ber");
84 * // After successful initialization:
85 * // - game.mlx contains valid MLX pointer
86 * // - game.map contains validated map data
87 * // - game.map_width and game.map_height are set
88 */
89static void init_mlx_and_map(t_game *game, char *map_path)
90{
91 ft_printf("Initializing MLX...\n");
92 game->mlx = mlx_init();
93 if (!game->mlx)
94 {
95 cleanup_game(game);
96 ft_printf("Error: MLX initialization failed\n");
97 exit(1);
98 }
99 ft_printf("Reading map: %s\n", map_path);
100 game->map = read_map(map_path, &game->map_width, &game->map_height);
101 if (!game->map || !validate_map(game))
102 {
103 ft_printf("Error: Invalid map: %s\n", map_path);
104 cleanup_game(game);
105 exit(1);
106 }
107}

109/**
