15/**
16 * @brief Loads a sequence of sprite frames from files
17 * @param mlx MLX instance pointer
18 * @param arr Array to store the loaded sprites
19 * @param base_path Base path to the sprite files
20 * @param frames Number of frames to load
21 *
22 * @details
23 * Loads a sequence of XPM files named f0.xpm through f{frames-1}.xpm
24 * from the specified directory. Each sprite is scaled to 32x32 pixels.
25 */
26void load_sprite_array(t_game *game, void **arr, char *base_path, int frames)
27{
28 char path[256];
29 void *orig_img;
30 void *scaled_img;
31 int params[3];
32 int i;
33
34 i = 0;
35 while (i < frames)
36 {
37 ft_snprintf(path, sizeof(path), "%s/f%d.xpm", base_path, i);
38 orig_img = mlx_xpm_file_to_image(game->mlx, path, &params[0], &params[1]);
39 if (!orig_img)
40 handle_error(game, "Failed to load sprite");
41 scaled_img = scale_image(game->mlx, orig_img, params[0], params[1]);
42 mlx_destroy_image(game->mlx, orig_img);
43 if (!scaled_img)
44 handle_error(game, "Failed to scale image");
45 arr[i++] = scaled_img;
46 }
47}

49static void load_single_texture(t_game *game, void **dest, char *path)
50{
51 void *orig_img;
52
53 orig_img = mlx_xpm_file_to_image(game->mlx, path, &game->img_width,
54 &game->img_height);
55 if (!orig_img)
56 {
57 ft_printf("Error: Failed to load texture: %s\n", path);
58 exit(1);
59 }
60 *dest = scale_image(game->mlx, orig_img, game->img_width, game->img_height);
61 mlx_destroy_image(game->mlx, orig_img);
62 if (!*dest)
63 {
64 ft_printf("Error: Failed to scale texture: %s\n", path);
65 exit(1);
66 }
67}

69/**
70 * @brief Loads all game assets from files
71 * @param game Pointer to game structure
72 *
73 * @details
74 * Loads and initializes:
75 * - Floor tiles (10 variants)
76 * - Player animations (idle and running)
77 * - Collectible items
78 * - Wall textures

83{

87 ft_printf("Loading assets...\n");

92 6);
93 load_sprite_array(game, game->collect, "./final_assets/chest", 8);
94 load_single_texture(game, &game->wall, "./final_assets/wall/f1.xpm");

97 ft_printf("All assets loaded successfully!\n");
98}
