15/**
16 * @brief Initializes random floor tile variations
17 * @param game Pointer to game structure
18 *
19 * @details
20 * Creates a 2D array of floor type indices.
21 * Each position gets a random floor variant (0-9)
22 * for visual variety in the game.
23 */
24void init_floor_types(t_game *game)
25{
26 int i;
27 int j;
28
29 i = 0;
30 game->floor_types = (unsigned char **)malloc(sizeof(unsigned char *)
31 * game->map_height);
32 while (i < game->map_height)
33 {
34 j = 0;
35 game->floor_types[i] = (unsigned char *)malloc(sizeof(unsigned char)
36 * game->map_width);
37 while (j < game->map_width)
38 {
39 game->floor_types[i][j] = (unsigned char)(rand() % 10);
40 j++;
41 }
42 i++;
43 }
44}

46/**
47 * @brief Locates and sets the player's starting position
48 * @param game Pointer to game structure
49 *
50 * @details
51 * Scans the map for 'P' character and sets
52 * player_x and player_y coordinates accordingly.
53 */
54void init_player_position(t_game *game)
55{
56 int i;
57 int j;
58
59 i = 0;
60 while (game->map[i])
61 {
62 j = 0;
63 while (game->map[i][j])
64 {
65 if (game->map[i][j] == 'P')
66 {
67 game->player_x = j;
68 game->player_y = i;
69 return ;
70 }
71 j++;
72 }
73 i++;
74 }
75}
