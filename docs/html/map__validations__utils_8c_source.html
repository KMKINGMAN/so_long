20int check_map_borders(t_game *game)
21{
22 int j;
23
24 if (game->map_height < 3 || game->map_width < 3)
25 return (ft_printf("Error: Map too small\n") && 0);
26 j = -1;
27 while (++j < game->map_width)
28 if (game->map[0][j] != '1' || game->map[game->map_height - 1][j] != '1')
29 return (ft_printf("Error: Map not surrounded by walls\n") && 0);
30 return (1);
31}

33int check_elements(t_game *game, int i, int j, t_counts *count)
34{
35 char c;
36
37 c = game->map[i][j];
38 if (c == 'P')

40 count->player++;

57int check_map_content(t_game *game, t_counts *count)
58{
59 int i;
60 int j;
61
62 i = -1;
63 while (++i < game->map_height)
64 {
65 if ((int)ft_strlen(game->map[i]) != game->map_width)
66 return (ft_printf("Error: Map not rectangular\n") && 0);
67 if (game->map[i][0] != '1' || game->map[i][game->map_width - 1] != '1')
68 return (ft_printf("Error: Map not surrounded by walls\n") && 0);
69 j = -1;
70 while (++j < game->map_width)
71 if (!check_elements(game, i, j, count))
72 return (0);
73 }
74 return (1);
