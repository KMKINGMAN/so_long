52/**
53 * @brief Handles player movement and collision
54 *
55 * @param game Pointer to game structure
56 * @param dx X-axis movement (-1, 0, 1)
57 * @param dy Y-axis movement (-1, 0, 1)
58 *
59 * @details
60 * - Updates player position if move is valid
61 * - Handles collectible pickup
62 * - Manages exit condition
63 * - Updates player direction and animation state
64 */
65void move_player(t_game *game, int dx, int dy)
66{
67 int new_x;
68 int new_y;
69
70 new_x = game->player_x + dx;
71 new_y = game->player_y + dy;
72 if (dx > 0)
73 game->facing_right = 1;
74 else if (dx < 0)
75 game->facing_right = 0;
76 if (game->map[new_y][new_x] == '1')
77 return;
78 if (game->map[new_y][new_x] == 'C')
79 handle_collectible(game);
80 if (game->map[new_y][new_x] == 'E')
81 {
82 handle_exit(game);
83 return;
84 }
85 update_player_pos(game, new_x, new_y);
86}

88/**
89 * @brief Handles keyboard input events
90 * @param keycode The key code of the pressed key
91 * @param game Pointer to the game structure
92 * @return int Always returns 0
93 *
94 * @details Controls:
95 * - ESC (65307): Exits the game
96 * - W/Up (13/119): Move up
97 * - S/Down (1/115): Move down
98 * - A/Left (0/97): Move left
99 * - D/Right (2/100): Move right
100 */
101int key_hook(int keycode, t_game *game)
102{
103 if (keycode == 65307) // ESC key
104 {
105 cleanup_game(game);
106 exit(0);
107 }
108 else if (keycode == 13 || keycode == 119)
109 move_player(game, 0, -1);
110 else if (keycode == 1 || keycode == 115)
111 move_player(game, 0, 1);
112 else if (keycode == 0 || keycode == 97)
113 move_player(game, -1, 0);
114 else if (keycode == 2 || keycode == 100)
115 move_player(game, 1, 0);
116 return (0);
117}
