20/**
21 * @brief Scales an image to a new size
22 *
23 * @param mlx MLX instance pointer
24 * @param original Original image
25 * @param width Original width
26 * @param height Original height
27 * @param new_size Target size
28 * @return void* Scaled image pointer or NULL on failure
29 *
30 * @details
31 * Creates a new scaled version of the input image using
32 * nearest neighbor interpolation.
33 */
34void *scale_image(void *mlx, void *original, int width, int height, int n_size)
35{
36 void *new_img;
37 int *orig_data;
38 int *new_data;
39 int params[5];
40
41 new_img = mlx_new_image(mlx, n_size, n_size);
42 if (!new_img)
43 return (NULL);
44 orig_data = (int *)mlx_get_data_addr(original, &params[0], &params[1],
45 &params[2]);
46 if (!orig_data)
47 {
48 mlx_destroy_image(mlx, new_img);
49 return (NULL);
50 }
51 new_data = (int *)mlx_get_data_addr(new_img, &params[0], &params[1],
52 &params[2]);
53 if (!new_data)
54 {
55 mlx_destroy_image(mlx, new_img);
56 return (NULL);
57 }
58 params[3] = -1;
59 while (++params[3] < n_size)
60 {
61 params[4] = -1;
62 while (++params[4] < n_size)
63 if (params[3] * width / n_size * width + params[4] * width / n_size >= 0 && params[3] * width / n_size * width + params[4] * width / n_size < (width * height))
64 new_data[params[3] * n_size + params[4]] = orig_data[params[3] * width / n_size * width + params[4] * width / n_size];
65 }
66 return (new_img);
67}

69/**
70 * @brief Creates a horizontally flipped copy of an image
71 * @param mlx MLX instance pointer
72 * @param img Source image to flip
73 * @return void* Pointer to the new flipped image
74 *
75 * @details
76 * Creates a new image that is a mirror of the input image.
77 * Used for creating left-facing sprites from right-facing ones.
78 * Maintains transparency information.
79 */
80void *flip_image_horizontal(void *mlx, void *img)
81{
82 int width;
83 int height;
84 int params[3];
85 void *new_img;
86 char *data[2];
87
88 width = 32;
89 height = 32;
90 new_img = mlx_new_image(mlx, width, height);
91 data[0] = mlx_get_data_addr(new_img, &params[0], &params[1], &params[2]);
92 data[1] = mlx_get_data_addr(img, &params[0], &params[1], &params[2]);
93 params[0] = -1;
94 while (++params[0] < height)
95 {
96 params[2] = -1;
97 while (++params[2] < width)
98 {
99 *(unsigned int *)(data[0] + params[0] * params[1] + (width - 1 - params[2]) * 4) = *(unsigned int *)(data[1] + params[0] * params[1] + params[2] * 4);
100 }
101 }
102 return (new_img);
103}

105/**
106 * @brief Loads a sequence of sprite frames from files
107 * @param mlx MLX instance pointer
108 * @param arr Array to store the loaded sprites
109 * @param base_path Base path to the sprite files
110 * @param frames Number of frames to load
111 *
112 * @details
113 * Loads a sequence of XPM files named f0.xpm through f{frames-1}.xpm
114 * from the specified directory. Each sprite is scaled to 32x32 pixels.
115 */
116void load_sprite_array(void *mlx, void **arr, char *base_path, int frames)
117{
118 char path[256];
119 void *orig_img;
120 void *scaled_img;
121 int params[3];
122 int i;
123
124 i = 0;
125 while (i < frames)
126 {
127 ft_snprintf(path, sizeof(path), "%s/f%d.xpm", base_path, i);
128 orig_img = mlx_xpm_file_to_image(mlx, path, &params[0], &params[1]);
129 if (!orig_img)
130 {
131 ft_printf("Error: Failed to load %s\n", path);
132 exit(1);
133 }
134 scaled_img = scale_image(mlx, orig_img, params[0], params[1], 32);
135 mlx_destroy_image(mlx, orig_img);
136 if (!scaled_img)
137 {
138 ft_printf("Error: Failed to scale image %s\n", path);
139 exit(1);
140 }
141 arr[i] = scaled_img;
142 i++;
143 }
144}

146/**
147 * @brief Creates left-facing versions of all player sprites
148 * @param game Pointer to game structure
149 *
150 * @details
151 * Creates mirrored versions of both idle and running animations
152 * for the player character. These are used when the player is
153 * moving or facing left.
154 */
155void flip_sprites(t_game *game)
156{
157 int i;
158
159 i = 0;
160 while (i < 6)
161 {
162 game->player_idle_left[i] = flip_image_horizontal(game->mlx,
163 game->player_idle[i]);
164 game->player_run_left[i] = flip_image_horizontal(game->mlx,
165 game->player_run[i]);
166 i++;
167 }
168}

170static void load_single_texture(t_game *game, void **dest, char *path)
171{
172 void *orig_img;
173
174 orig_img = mlx_xpm_file_to_image(game->mlx, path,
175 &game->img_width, &game->img_height);
176 if (!orig_img)
177 {
178 ft_printf("Error: Failed to load texture: %s\n", path);
179 exit(1);
180 }
181 *dest = scale_image(game->mlx, orig_img, game->img_width,
182 game->img_height, game->tile_size);
183 mlx_destroy_image(game->mlx, orig_img);
184 if (!*dest)
185 {
186 ft_printf("Error: Failed to scale texture: %s\n", path);
187 exit(1);
188 }
189}

220}
