28/**
29 * Processes image scaling by mapping original image data to a new scaled format
30 *
31 * @param new_data Pointer to destination
32 * array for scaled image data
33 * @param orig_data Pointer to source array
34 * containing original image data
35 * @param p Pointer to array containing
36 * scaling parameters where p[4] represents height
37 * @param width Width of the original image
38 *
39 * @details
40 * The function scales the image to a fixed size of 32x32 pixels (n_size).
41 * For each pixel in the new scaled image, it maps back to corresponding pixel
42 * in original image using proportional scaling calculations.
43 *
44 * @example
45 * int original[100 * 100]; // Original 100x100 image
46 * int scaled[32 * 32]; // Will contain 32x32 scaled image
47 * int params[5] = {0, 0, 0, 0, 100}; // params where p[4] is height
48 * process_scaling(scaled, original, params, 100);
49 *
50 * @note
51 * - Output is always scaled to 32x32 pixels
52 * - Function includes bounds checking to prevent buffer overflows
53 * - Scaling is done using integer arithmetic for performance
54 */
55static void process_scaling(int *new_data, int *orig_data, int *p, int width)
56{
57 int i;
58 int j;
59 int n_size;
60
61 i = -1;
62 n_size = 32;
63 while (++i < n_size)
64 {
65 j = -1;
66 while (++j < n_size)
67 {
68 if (i * width / n_size * width + j * width / n_size >= 0
69 && i * width / n_size * width + j * width / n_size
70 < (width * p[4]))
71 new_data[i * n_size + j] = orig_data[i * width / n_size
72 * width + j * width / n_size];
73 }
74 }
75}

77/**
78 * @brief Scales an MLX image to a fixed size of 32x32 pixels
79 *
80 * This function takes an original
81 * MLX image and creates a new scaled version
82 * with dimensions of 32x32 pixels.
83 * It handles the memory allocation and pixel
84 * data transfer between the images.
85 *
86 * @param mlx Pointer to the MLX instance
87 * @param original Pointer to the original image to be scaled
88 * @param width Original image width in pixels
89 * @param height Original image height in pixels
90 *
91 * @return Pointer to the newly created scaled image,
92 * or NULL if an error occurred
93 *
94 * @note The function internally uses create_scaled_image()
95 * and process_scaling()
96 * to handle the actual scaling operation
97 *
98 * @example
99 * void *mlx = mlx_init();
100 * void *original_img = mlx_xpm_file_to_image(mlx, "sprite.xpm",
101 * &width, &height);
102 * void *scaled_img = scale_image(mlx, original_img,
103 * width, height);
104 * if (scaled_img)
105 * mlx_put_image_to_window(mlx, window, scaled_img, x, y);
106 */
107void *scale_image(void *mlx, void *original, int width, int height)
108{
109 void *new_img;
110 int *orig_data;
111 int *new_data;
112 int params[5];
113
114 new_img = create_scaled_image(mlx, 32);
115 if (!new_img)
116 return (NULL);
117 orig_data = (int *)mlx_get_data_addr(original, &params[0], &params[1],
118 &params[2]);
119 if (!orig_data)
120 {
121 mlx_destroy_image(mlx, new_img);
122 return (NULL);
123 }
124 new_data = (int *)mlx_get_data_addr(new_img, &params[0], &params[1],
125 &params[2]);
126 if (!new_data)
127 {
128 mlx_destroy_image(mlx, new_img);
129 return (NULL);
130 }
131 params[4] = height;
132 process_scaling(new_data, orig_data, params, width);
133 return (new_img);
134}

136/**
137 * @brief Creates a horizontally flipped copy of an image
138 * @param game to get MLX instance pointer
139 * @param img Source image to flip
140 * @return void* Pointer to the new flipped image
141 *
142 * @details
143 * Creates a new image that is a mirror of the input image.
144 * Used for creating left-facing sprites from right-facing ones.
145 * Maintains transparency information.
146 */
147void *flip_image_horizontal(t_game *game, void *img)
148{
149 int params[3];
150 void *new_img;
151 char *data[2];
152
153 new_img = mlx_new_image(game->mlx, 32, 32);
154 data[0] = mlx_get_data_addr(new_img, &params[0], &params[1], &params[2]);
155 data[1] = mlx_get_data_addr(img, &params[0], &params[1], &params[2]);
156 if(!data[0] || !data[1])
157 {
158 mlx_destroy_image(game->mlx, new_img);
159 handle_error(game, "Faild to load the image data");
160 }
161 params[0] = -1;
162 while (++params[0] < 32)
163 {
164 params[2] = -1;
165 while (++params[2] < 32)
166 {
167 *(unsigned int *)(data[0] + params[0] * params[1] + (32 - 1
168 - params[2]) * 4) = *(unsigned int *)(data[1]
169 + params[0] * params[1] + params[2] * 4);
170 }
171 }
172 return (new_img);
173}
