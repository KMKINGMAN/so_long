21/**
22 * @brief Copies a sprite to the buffer image with transparency handling
23 *
24 * @param dst Destination buffer image
25 * @param src_ptr Source sprite pointer
26 * @param x X coordinate in destination
27 * @param y Y coordinate in destination
28 *
29 * @details
30 * Copies a 32x32 sprite to the buffer while handling transparency.
31 * Only non-transparent pixels (alpha != 0) are copied.
32 */
33static void put_img_to_img_part(t_img *dest, t_img *sprite, t_pos pos)
34{
35 int sprite_pos;
36 int dest_pos;
37
38 sprite_pos = pos.col * (sprite->bpp / 8) + pos.row * sprite->line_len;
39 dest_pos = (pos.dest_x + pos.col) * (dest->bpp / 8)
40 + (pos.dest_y + pos.row) * dest->line_len;
41 if ((*(unsigned int *)(sprite->addr + sprite_pos) & 0xFF000000) == 0)
42 *(unsigned int *)(dest->addr + dest_pos)
43 = *(unsigned int *)(sprite->addr + sprite_pos);
44}

46void put_img_to_img(t_img *dest, void *sprite_ptr, int dest_x, int dest_y)
47{
48 t_img sprite;
49 t_pos pos;
50
51 sprite.img_ptr = sprite_ptr;
52 sprite.addr = mlx_get_data_addr(sprite.img_ptr, &sprite.bpp,
53 &sprite.line_len, &sprite.endian);
54 pos.row = 0;
55 pos.dest_x = dest_x;
56 pos.dest_y = dest_y;
57 while (pos.row < 32)
58 {
59 pos.col = 0;
60 while (pos.col < 32)
61 {
62 put_img_to_img_part(dest, &sprite, pos);
63 pos.col++;
64 }
65 pos.row++;
66 }
67}

90/**
91 * @brief Handles game animation frames and state updates
92 *
93 * @param game Pointer to game structure
94 * @return int Always returns 0
95 *
96 * @details
97 * - Updates animation frames every 100 delay ticks
98 * - Manages player running/idle state transitions
99 * - Triggers frame rendering
100 */
101int animate(t_game *game)
102{
103 static struct timespec last_time;
104 struct timespec current_time;
105 double elapsed;
106
107 clock_gettime(CLOCK_MONOTONIC, &current_time);
108 elapsed = (current_time.tv_sec - last_time.tv_sec) * 1000.0 +
109 (current_time.tv_nsec - last_time.tv_nsec) / 1000000.0;
110 if (elapsed >= 100.0 || last_time.tv_sec == 0)
111 {
112 game->frame = (game->frame + 1) % 6;
113 last_time = current_time;
114
115 if (game->is_running)
116 {
117 game->run_counter++;
118 if (game->run_counter >= 20)
119 {
120 game->is_running = 0;
121 game->run_counter = 0;
122 }
123 }
124 }
125 render_frame(game);
126 return (0);
127}

129/**
130 * @brief Renders a complete frame of the game
131 * @param game Pointer to game structure
132 * @return int Always returns 0
133 *
134 * @details Rendering order:
135 * 1. Clear buffer
136 * 2. Draw floor tiles
137 * 3. Draw exit
138 * 4. Draw walls
139 * 5. Draw player with direction/animation
140 * 6. Draw collectibles
141 * 7. Draw UI elements
142 */
143static void draw_floor_and_exit(t_game *game)
144{
145 int i;
146 int j;
147 int x;
148 int y;
149
150 i = 0;
151 while (game->map[i])
152 {
153 j = 0;
154 while (game->map[i][j])
155 {
156 x = j * game->tile_size;
157 y = i * game->tile_size;
158 put_img_to_img(&game->buffer,
159 game->floor[game->floor_types[i][j]], x, y);
160 if (game->map[i][j] == 'E' && game->can_exit)
161 put_img_to_img(&game->buffer, game->exit, x, y);
162 j++;
163 }
164 i++;
165 }
166}

168static void draw_player(t_game *game, int x, int y)
169{
170 void *sprite;
171
172 if (game->facing_right)
173 {
174 if (game->is_running)
175 sprite = game->player_run[game->frame % 6];
176 else
177 sprite = game->player_idle[game->frame % 6];
178 }
179 else
180 {
181 if (game->is_running)
182 sprite = game->player_run_left[game->frame % 6];
183 else
184 sprite = game->player_idle_left[game->frame % 6];
185 }
186 put_img_to_img(&game->buffer, sprite, x, y);
187}

192 int j;

194 int y;

203 y = i * game->tile_size;
204 if (game->map[i][j] == '1')
205 put_img_to_img(&game->buffer, game->wall, x, y);
206 else if (game->map[i][j] == 'P')
207 draw_player(game, x, y);
208 else if (game->map[i][j] == 'C')
209 put_img_to_img(&game->buffer,
210 game->collect[game->frame % 8], x, y);

217int render_frame(t_game *game)
218{
219 char moves_str[32];
220
221 ft_memset(game->buffer.addr, 0,
222 game->map_width * game->tile_size * game->map_height
223 * game->tile_size * (game->buffer.bpp / 8));
224 draw_floor_and_exit(game);
225 draw_entities(game);
226 mlx_put_image_to_window(game->mlx, game->win,
227 game->buffer.img_ptr, 0, 0);
228 ft_snprintf(moves_str, sizeof(moves_str), "Moves: %d", game->moves_count);
229 mlx_string_put(game->mlx, game->win, 10, 20, 0xFFFFFF, moves_str);
230 return (0);
231}
