20/**
21 * @brief Copies a sprite to the buffer image with transparency handling
22 *
23 * @param dst Destination buffer image
24 * @param src_ptr Source sprite pointer
25 * @param x X coordinate in destination
26 * @param y Y coordinate in destination
27 *
28 * @details
29 * Copies a 32x32 sprite to the buffer while handling transparency.
30 * Only non-transparent pixels (alpha != 0) are copied.
31 */
32static void put_img_to_img_part(t_img *dest, t_img *sprite, t_pos pos)
33{
34 int sprite_pos;
35 int dest_pos;
36
37 sprite_pos = pos.col * (sprite->bpp / 8) + pos.row * sprite->line_len;
38 dest_pos = (pos.dest_x + pos.col) * (dest->bpp / 8) + \
39 (pos.dest_y + pos.row) * dest->line_len;
40 if ((*(unsigned int *)(sprite->addr + sprite_pos) & 0xFF000000) == 0)
41 *(unsigned int *)(dest->addr
42 + dest_pos) = *(unsigned int *)(sprite->addr + sprite_pos);
43}

45void put_img_to_img(t_img *dest, void *sprite_ptr, int dest_x, int dest_y)
46{
47 t_img sprite;
48 t_pos pos;
49
50 sprite.img_ptr = sprite_ptr;
51 sprite.addr = mlx_get_data_addr(sprite.img_ptr, &sprite.bpp,
52 &sprite.line_len, &sprite.endian);
53 pos.row = 0;
54 pos.dest_x = dest_x;
55 pos.dest_y = dest_y;
56 while (pos.row < 32)
57 {
58 pos.col = 0;
59 while (pos.col < 32)
60 {
61 put_img_to_img_part(dest, &sprite, pos);
62 pos.col++;
63 }
64 pos.row++;
65 }
66}

89/**
90 * @brief Handles game animation frames and state updates
91 *
92 * @param game Pointer to game structure
93 * @return int Always returns 0
94 *
95 * @details
96 * - Updates animation frames every 100 delay ticks
97 * - Manages player running/idle state transitions
98 * - Triggers frame rendering
99 */
100int animate(t_game *game)
101{
102 static struct timespec last_time;
103 struct timespec current_time;
104 double elapsed;
105
106 clock_gettime(CLOCK_MONOTONIC, &current_time);
107 elapsed = (current_time.tv_sec - last_time.tv_sec) * 1000.0
108 + (current_time.tv_nsec - last_time.tv_nsec) / 1000000.0;
109 if (elapsed >= 100.0 || last_time.tv_sec == 0)
110 {
111 game->frame = (game->frame + 1) % 6;
112 last_time = current_time;
113 if (game->is_running)
114 {
115 game->run_counter++;
116 if (game->run_counter >= 20)
117 {
118 game->is_running = 0;
119 game->run_counter = 0;
120 }
121 }
122 }
123 render_frame(game);
124 return (0);
125}
