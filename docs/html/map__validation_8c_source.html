20static int check_map_borders(t_game *game)
21{
22 int j;
23
24 if (game->map_height < 3 || game->map_width < 3)
25 return (ft_printf("Error: Map too small\n") && 0);
26 for (j = 0; j < game->map_width; j++)
27 if (game->map[0][j] != '1' || game->map[game->map_height - 1][j] != '1')
28 return (ft_printf("Error: Map not surrounded by walls\n") && 0);
29 return (1);
30}

32static int check_elements(t_game *game, int i, int j, t_counts *count)
33{
34 char c;
35
36 c = game->map[i][j];
37 if (c == 'P')
38 {
39 count->player++;
40 game->player_x = j;
41 game->player_y = i;
42 }
43 else if (c == 'E')
44 {
45 count->exit++;
46 game->exit_x = j;
47 game->exit_y = i;
48 }
49 else if (c == 'C')
50 count->collect++;
51 else if (c != '0' && c != '1')
52 return (ft_printf("Error: Invalid character in map\n") && 0);

68int validate_map(t_game *game)
69{
70 t_counts count;
71 int i;
72 int j;
73
74 count = (t_counts){0, 0, 0};
75 if (!check_map_borders(game))
76 return (0);
77 i = -1;
78 while (++i < game->map_height)
79 {
80 if ((int)ft_strlen(game->map[i]) != game->map_width)
81 return (ft_printf("Error: Map not rectangular\n") && 0);
82 if (game->map[i][0] != '1' || game->map[i][game->map_width - 1] != '1')
83 return (ft_printf("Error: Map not surrounded by walls\n") && 0);
84 j = -1;
85 while (++j < game->map_width)
86 if (!check_elements(game, i, j, &count))
87 return (0);
88 }
89 if (!validate_counts(game, count))
90 return (0);
91 ft_printf("Map validation successful:\n");
92 ft_printf("- Size: %dx%d\n", game->map_width, game->map_height);
93 ft_printf("- Collectibles: %d\n", count.collect);
94 return (1);
95}
